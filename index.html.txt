    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const consentCheckbox = document.getElementById('consentCheckbox');
            const termsCheckbox = document.getElementById('termsCheckbox');
            const shareBtn = document.getElementById('shareBtn');
            const clickCountElement = document.getElementById('clickCount');
            const successMessage = document.getElementById('successMessage');
            let clickCount = 0;
            
            // WhatsApp শেয়ার লিঙ্ক
            const message = encodeURIComponent("❤️ আপনার জন্য একটি বিশেষ বার্তা আছে! জীবন সুন্দর, ভালোবাসা ছড়িয়ে দিন! 💌");
            const whatsappURL = `https://wa.me/?text=${message}`;
            
            // অনুমতি checkbox পরিবর্তন শনাক্ত করুন
            function updateShareButton() {
                if(consentCheckbox.checked && termsCheckbox.checked) {
                    shareBtn.disabled = false;
                    shareBtn.style.opacity = "1";
                    shareBtn.style.cursor = "pointer";
                } else {
                    shareBtn.disabled = true;
                    shareBtn.style.opacity = "0.7";
                    shareBtn.style.cursor = "not-allowed";
                }
            }
            
            consentCheckbox.addEventListener('change', updateShareButton);
            termsCheckbox.addEventListener('change', updateShareButton);
            
            shareBtn.addEventListener('click', function() {
                if(!consentCheckbox.checked || !termsCheckbox.checked) {
                    alert("দয়া করে প্রথমে সবগুলো অনুমতি checkbox টি চেক করুন");
                    return;
                }
                
                clickCount++;
                clickCountElement.textContent = clickCount;
                
                // সাফল্য বার্তা দেখান
                successMessage.style.display = 'block';
                
                // WhatsApp ওপেন করুন
                setTimeout(function() {
                    window.open(whatsappURL, '_blank');
                }, 1000);
                
                // Create some simple confetti effect
                createConfetti();
            });
            
            function createConfetti() {
                const confettiCount = 30;
                const container = document.querySelector('.container');
                
                for (let i = 0; i < confettiCount; i++) {
                    const confetti = document.createElement('div');
                    confetti.className = 'confetti';
                    confetti.style.left = Math.random() * 100 + '%';
                    confetti.style.top = '-10px';
                    confetti.style.backgroundColor = getRandomColor();
                    confetti.style.transform = `scale(${Math.random() * 0.5 + 0.5})`;
                    container.appendChild(confetti);
                    
                    // Animate confetti
                    const animationDuration = Math.random() * 3000 + 2000;
                    confetti.animate([
                        { top: '-10px', transform: 'rotate(0deg)' },
                        { top: '100%', transform: `rotate(${Math.random() * 360}deg)` }
                    ], {
                        duration: animationDuration,
                        easing: 'cubic-bezier(0.1, 0.8, 0.3, 1)',
                        fill: 'forwards'
                    });
                    
                    // Remove confetti after animation
                    setTimeout(() => {
                        confetti.remove();
                    }, animationDuration);
                }
            }
            
            function getRandomColor() {
                const colors = ['#ff6b6b', '#4ecdc4', '#45b7d1', '#f9c74f', '#e91e63', '#9c27b0'];
                return colors[Math.floor(Math.random() * colors.length)];
            }
        });
    </script>
